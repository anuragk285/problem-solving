# find minimum in rotated sorted array

```python
class Solution(object):
    def findMin(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        res = nums[0]
        l, r = 0, len(nums)-1

        while l <= r:
            mid = (l+r)//2
            res = min(res, nums[mid])
            if nums[l] <= nums[r]:
                res = min(res, nums[l])
                break
            elif nums[mid] >= nums[l] and nums[mid] >= nums[r]:
                l = mid+1
            else : r =mid-1
        return res
```

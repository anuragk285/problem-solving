# copy list with random pointer

```python
"""
# Definition for a Node.
class Node:
    def __init__(self, x, next=None, random=None):
        self.val = int(x)
        self.next = next
        self.random = random
"""

class Solution(object):
    def copyRandomList(self, head):
        """
        :type head: Node
        :rtype: Node
        """
        hash_map = {None : None}

        p = head
        while p:
            hash_map[p] = Node(p.val)
            p = p.next
        
        p =head
        ans_head = None
        while p:
            if not ans_head:
                ans_head = hash_map[p]
                ans_head.next = hash_map[p.next]
                ans_head.random = hash_map[p.random]
            else:
                q = hash_map[p]
                q.next = hash_map[p.next]
                q.random = hash_map[p.random]
            p =p.next
        return ans_head
```

# Minimum operations to make integer zero

```python
class Solution(object):
    def makeTheIntegerZero(self, num1, num2):
        """
        :type num1: int
        :type num2: int
        :rtype: int
        """
        def is_possible(x):
            diff = num1-(x*num2)
            if diff <= 0: return False
            min_operations = str(format(diff, "b")).count('1')
            max_operations = diff
            if min_operations <= x <= max_operations: return True
            return False

        if num1 == 0: return 0
        if num1 <= num2 : return -1

        for k in range(1, 100):
            if is_possible(k): return k
        
        return -1
```

# four divisors
```c++
class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {

        int sum =0;

        for(int i=0; i<nums.size(); i++){

            int divisors =1, divSum =0;

            for(int j=1; j<nums[i]/2+1; j++){

                if(nums[i]%j == 0){
                    divisors++;
                    divSum += j;
                }
                if(divisors > 4) break;
            }
            divSum += nums[i];
            cout<<"n = "<<nums[i]<<", divisors = "<<divisors<<endl;

            if(divisors == 4) sum += divSum;
        }
        return sum;
    }
```

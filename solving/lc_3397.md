# Minimum number of distinct elements after operation

```python
class Solution(object):
    def maxDistinctElements(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: int
        """
        nums.sort()
        ex_count = 0
        ans = []

        for n in nums:
            if not ans:
                ans.append(n-k)
            else:
                if ans[-1] < n-k:
                    ans.append(n-k)
                elif ans[-1]+1 <= n+k:
                    ans.append(ans[-1]+1)
                else:
                    ex_count += 1
        
        return len(nums) - ex_count
```

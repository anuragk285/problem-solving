# Minimum Score Triangulation of polygon

```python
class Solution(object):
    def minScoreTriangulation(self, values):
        """
        :type values: List[int]
        :rtype: int
        """
        n = len(values)
        self.dp = [[-1]*n for _ in range(n)]

        def recursion(i, j):
            if i+1 == j or i==j:
                return 0    
            if self.dp[i][j] != -1:
                return self.dp[i][j]
            ans = float("inf")
            for k in range(i+1, j):
                left = recursion(i, k)
                right = recursion(k, j)
                current_tri = values[i] * values[k] * values[j]
                ans = min(ans, left + current_tri + right)
            self.dp[i][j] = ans
            return self.dp[i][j]
        
        return recursion(0, n-1)
```

# BItwise xor of al pairings 

```c++
class Solution {
public:
    int xorAllNums(vector<int>& nums1, vector<int>& nums2) {
        
        int xor1 = 0, xor2 = 0, ans = 0;

        for(int i : nums1) xor1 ^= i;

        for(int j : nums2) xor2 ^= j;

        if(nums2.size() % 2 == 1) ans ^= xor1;

        if(nums1.size() % 2 == 1) ans ^= xor2;

        return ans;
    }
};
```

class ExamTracker(object):

    def __init__(self):
        self.times = []
        self.scores = []
        self.score_upto = []

    def record(self, time, score):
        """
        :type time: int
        :type score: int
        :rtype: None
        """
        self.times.append(time)
        self.scores.append(score)
        if len(self.score_upto) == 0:
            self.score_upto.append(score)
        else:
            self.score_upto.append(self.score_upto[-1] + score)

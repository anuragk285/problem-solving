# Design movie rental system

```python
from sortedcontainers import SortedSet
class MovieRentingSystem(object):

    def __init__(self, n, entries):
        """
        :type n: int
        :type entries: List[List[int]]
        """
        self.prices = defaultdict(SortedSet)
        self.unrented = defaultdict(SortedSet)
        self.rented = SortedSet()
        for s, m, p in entries:
            self.prices[(s, m)] = p
            self.unrented[m].add((p, s))
        

    def search(self, movie):
        """
        :type movie: int
        :rtype: List[int]
        """
        ans = []
        for price, shop in self.unrented[movie]:
            ans.append(shop)
            if len(ans) == 5:
                break
        return ans

    def rent(self, shop, movie):
        """
        :type shop: int
        :type movie: int
        :rtype: None
        """
        price = self.prices[(shop, movie)]
        self.unrented[movie].remove((price, shop))
        self.rented.add((price, shop, movie))
        

    def drop(self, shop, movie):
        """
        :type shop: int
        :type movie: int
        :rtype: None
        """
        price = self.prices[(shop, movie)]
        self.rented.remove((price, shop, movie))
        self.unrented[movie].add((price, shop))

    def report(self):
        """
        :rtype: List[List[int]]
        """
        ans = []
        for price, shop, movie in self.rented:
            ans.append([shop, movie])
            if len(ans) == 5:
                break
        return ans
```

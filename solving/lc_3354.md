# Make array elements equal to zero

```python
class Solution(object):
    def countValidSelections(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """ 
        self.nonZeros = sum(1 for n in nums if n > 0)
        ans = 0
        def simulate(temp, i, direction):
            curr = i
            nz = self.nonZeros
            while nz > 0 and 0 <= curr < len(temp):
                if temp[curr] > 0:
                    temp[curr] -= 1
                    if temp[curr] == 0:
                        nz -= 1
                    direction *= -1
                curr += direction
            return True if nz == 0 else False

        for i in range(len(nums)):
            if nums[i] == 0:
                if simulate(nums[:], i, 1): ans += 1
                if simulate(nums[:], i, -1): ans += 1

        return ans
```

# Longest Fabinocci substring

```python
class Solution(object):
    def longestSubarray(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        ans = 0
        length = 2
        n = len(nums)
        for i in range(2, n):
            if nums[i] == (nums[i-1] + nums[i-2]):
                if length == 0:
                    length = 3
                else:
                    length += 1
            else:
                ans = max(ans, length)
                length = 0
        return max(ans, length)
```

# Count number of Trapezoids I

```java
class Solution {
    public int countTrapezoids(int[][] points) {
        HashMap<Integer, Integer> lines = new HashMap<>();
        for(int[] point: points){
            lines.put(point[1], lines.getOrDefault(point[1], 0)+1);
        }
        long ans = 0, sum = 0;
        int mod = (int)1_000_000_007L;
        long[] freq = new long[lines.size()];
        int i=0;
        for(Map.Entry line : lines.entrySet()){
            long val = (long)((int)line.getValue());
            freq[i++] = (long)(val*(val-1)/2);
        }
        sum += freq[0];
        for(i=1; i<freq.length; i++){
            ans += (sum*freq[i]) % mod;
            sum += freq[i];
        }
        return (int)(ans%mod);
    }
}
```

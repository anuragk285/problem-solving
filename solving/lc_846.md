# Hand of Straights

```python
class Solution(object):
    def isNStraightHand(self, hand, groupSize):
        """
        :type hand: List[int]
        :type groupSize: int
        :rtype: bool
        """
        freq = {}
        for n in hand:
            freq[n] = freq.get(n, 0) + 1
        hand.sort()
        
        for i in hand:
            can_make = True
            for j in range(i, i+groupSize):
                if j not in freq or freq[j] <= 0:
                    can_make = False
                    break
            if can_make:
                for j in range(i, i+groupSize):
                    freq[j] -= 1
        
        for f in freq:
            if freq[f] > 0:
                return False
        
        return True
```

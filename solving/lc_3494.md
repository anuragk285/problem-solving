# Find the minimum amount of Time to brew potions

```python
class Solution(object):
    def minTime(self, skill, mana):
        """
        :type skill: List[int]
        :type mana: List[int]
        :rtype: int
        """
        n, m = len(skill), len(mana)
        times = [0]*n

        for i in range(n):
            if i == 0:
                times[i] = skill[i]*mana[0]
            else:
                times[i] = times[i-1] + skill[i]*mana[0]

        for i in range(1, m):
            for j in range(n):
                if j == 0:
                    times[j] = times[j] + skill[j]*mana[i]
                else:
                    times[j] = max(times[j-1], times[j]) + skill[j]*mana[i]
            
            for j in range(n-2, -1, -1):
                times[j] = times[j+1] - skill[j+1]*mana[i]
                
        return times[-1]
```

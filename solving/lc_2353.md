# Design a food rating system

```python
from sortedcontainers import SortedSet

class FoodRatings(object):

    def __init__(self, foods, cuisines, ratings):
        """
        :type foods: List[str]
        :type cuisines: List[str]
        :type ratings: List[int]
        """
        self.food_items = defaultdict(SortedSet) # cuisine : (-rating, food)
        self.food_type = {} # food : (cuisine, rating)
        for i in range(len(foods)):
            self.food_items[cuisines[i]].add((-ratings[i], foods[i]))
            self.food_type[foods[i]] = [cuisines[i], ratings[i]]
        

    def changeRating(self, food, newRating):
        """
        :type food: str
        :type newRating: int
        :rtype: None
        """
        cuisine, rating = self.food_type[food]
        self.food_items[cuisine].remove((-rating, food))
        self.food_items[cuisine].add((-newRating, food))
        self.food_type[food][1] = newRating
        

    def highestRated(self, cuisine):
        """
        :type cuisine: str
        :rtype: str
        """
        return self.food_items[cuisine][0][1]
```

# Sorrounded regions

```java
class Solution {
    char[][] board;
    int ROWS, COLS;
    boolean[][] visited;
    public void solve(char[][] board) {
        this.board = board;
        ROWS = board.length;
        COLS = board[0].length;
        visited = new boolean[ROWS][COLS];
        for(int i=0; i<ROWS; i++){
            if(board[i][0] == 'O' && !visited[i][0]) dfs(i, 0);
            if(board[i][COLS-1] == 'O' && !visited[i][COLS-1]) dfs(i, COLS-1);
        }
        for(int j=0; j<COLS; j++){
            if(board[0][j] == 'O' && !visited[0][j]) dfs(0, j);
            if(board[ROWS-1][j] == 'O' && !visited[ROWS-1][j]) dfs(ROWS-1, j);
        }
        for(int i=0; i<ROWS; i++){
            for(int j=0; j<COLS; j++){
                if(board[i][j] == 'O') board[i][j] = 'X';
                else if(board[i][j] == 'B') board[i][j] = 'O';
            }
        }
    }
    private void dfs(int r, int c){
        board[r][c] = 'B';
        visited[r][c] = true;
        int[] nRows = {r-1, r, r+1, r};
        int[] nCols = {c, c+1, c, c-1};
        for(int i=0; i<4; i++){
            r = nRows[i];
            c = nCols[i];
            if(isValid(r, c) && !visited[r][c] && board[r][c] == 'O'){
                dfs(r, c);
            }
        }
    }
    private boolean isValid(int r, int c){
        if(r<0 || r>=ROWS | c<0 || c>=COLS) return false;
        return true;
    }
}
```

# Minimum operations to transform string

```python
class Solution(object):
    def minOperations(self, s):
        """
        :type s: str
        :rtype: int
        """
        alpha = [0]*26
        for c in s:
            alpha[ord(c)-ord('a')] += 1
        count = 0
        for i in range(1, 25):
            if alpha[i] > 0:
                alpha[i+1] += alpha[i]
                alpha[i] = 0
                count += 1
        if alpha[25] > 0:
            alpha[0] += alpha[25]
            alpha[25] = 0
            count += 1
        return count
```

# Increment submatrices by One

```java
class Solution {
    public int[][] rangeAddQueries(int n, int[][] queries) {
        int[][] matrix = new int[n][n];
        for(int i=0; i<n; i++) for(int j=0; j<n; j++) matrix[i][j] = 0;
        for(int []q : queries){
            addOnetoMatrix(q[0], q[2], q[1], q[3], matrix);
        }
        return matrix;
    }
    public void addOnetoMatrix(int startRow, int endRow, int startCol, int endCol, int [][]mat){
        if(startRow >= mat.length || endRow < 0 || startCol >= mat[0].length || endCol < 0){
            return;
        }
        else{
            for(int i=startRow; i<=endRow; i++){
                for(int j=startCol; j<=endCol; j++){
                    mat[i][j]++;
                }
            }
        }
    }
}
```

# Sort array by diagonals

```python
class Solution(object):
    def sortMatrix(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: List[List[int]]
        """
        n = len(grid)

        for i in range(0, n):
            diag = []
            k = i
            for j in range(0, n-i):
                diag.append(grid[k][j])
                k += 1
            diag.sort(reverse=True)
            k = i
            for j in range(0, n-i):
                grid[k][j] = diag[j]
                k += 1

        for j in range(1, n):
            diag = []
            k = j
            for i in range(0, n-j):
                diag.append(grid[i][k])
                k +=1
            diag.sort()
            k = j
            for i in range(0, n-j):
                grid[i][k] = diag[i]
                k += 1
        return grid
```

# Sort Charaacters by Frequency

```java
class Solution {
    class Pair{
        char c = ' ';
        int f = 0;
        Pair(char c, int f){
            this.c = c;
            this.f = f;
        }
    }
    public String frequencySort(String s) {
        int[] sFreq = new int[26];
        int[] bFreq = new int[26];
        HashMap<Character, Integer> map = new HashMap<>();
        for(char c : s.toCharArray()){
            if('a' <= c && c <= 'z') sFreq[c-'a']++;
            else if('A' <= c && c <= 'Z') bFreq[c-'A']++;
            else map.put(c, map.getOrDefault(c, 0) + 1);
        }
        PriorityQueue<Pair> heap = new PriorityQueue<>(
            (a, b) -> Integer.compare(b.f, a.f)
        );
        for(int i=0; i<26; i++){
            char c = (char)('a' + i);
            heap.add(new Pair(c, sFreq[i]));
            c = (char)('A' + i);
            heap.add(new Pair(c, bFreq[i]));
        }
        for(Map.Entry<Character, Integer> entry : map.entrySet()){
            heap.add(new Pair(entry.getKey(), entry.getValue()));
        }
        String ans = "";
        while(!heap.isEmpty()){
            Pair p = heap.poll();
            ans += Character.toString(p.c).repeat(p.f);
        }
        return ans;
    }
}
```

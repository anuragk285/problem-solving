# Time based key value store

```python
class TimeMap(object):

    def __init__(self):
        self.time_map = {}

    def set(self, key, value, timestamp):
        """
        :type key: str
        :type value: str
        :type timestamp: int
        :rtype: None
        """
        if key in self.time_map : self.time_map[key].append((timestamp, value))
        
        else : self.time_map[key] = [(timestamp, value)]

    def get(self, key, timestamp):
        """
        :type key: str
        :type timestamp: int
        :rtype: str
        """
        if key not in self.time_map : return ""

        if self.time_map[key][0][0] > timestamp: return ""

        if self.time_map[key][-1][0] <= timestamp: return self.time_map[key][-1][1] 

        times = self.time_map.get(key, [])

        low, high = 0, len(times)-1
        res = ""
        while low <= high:
            mid = (low+high)//2
            if times[mid][0] <= timestamp:
                res = times[mid][1]
                low = mid+1
            else : high = mid-1

        return res


# Your TimeMap object will be instantiated and called as such:
# obj = TimeMap()
# obj.set(key,value,timestamp)
# param_2 = obj.get(key,timestamp)
```

# Number of laser beams in a bank

```python
class Solution(object):
    def numberOfBeams(self, bank):
        """
        :type bank: List[str]
        :rtype: int
        """
        projectors, receivers = 0, 0
        count = 0
        for i in range(len(bank)):
            if i == 0:
                projectors = bank[i].count('1')
            else:
                r = bank[i].count('1')
                if r == 0: continue
                count += projectors * r
                projectors = r
        return count
```

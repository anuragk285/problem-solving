# permutation in string

```python
class Solution(object):
    def checkInclusion(self, s1, s2):
        """
        :type s1: str
        :type s2: str
        :rtype: bool
        """
        if len(s1) > len(s2) : return False

        i, j = 0, 0
        t1 = [0]*26
        for c in s1: 
            t1[ord(c)-ord('a')] += 1
        
        t2 = [0]*26
        while j < len(s1):
            t2[ord(s2[j])-ord('a')] += 1
            j += 1
        if t1 == t2 : return True
        
        while i<len(s2) and j<len(s2):
            t2[ord(s2[i])-ord('a')] -= 1
            t2[ord(s2[j])-ord('a')] += 1
            if t1 == t2 : return True
            i, j = i+1, j+1
        
        return False
```

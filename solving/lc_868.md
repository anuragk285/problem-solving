# Binary gap

```python
class Solution(object):
    def binaryGap(self, n):
        """
        :type n: int
        :rtype: int
        """
        binary = ""
        while n>0:
            binary = str(n%2) + binary
            n //= 2
        
        indices = []
        for i in range(len(binary)) :
            if binary[i] == '1': 
                indices.append(i)

        max = 0; i = 1
        while i<len(indices):
            if (indices[i]-indices[i-1]) > max : max = (indices[i]-indices[i-1])
            i += 1
        
        return max
```

# Maximum subarray sum with length divisible by k

```python
class Solution(object):
    def maxSubarraySum(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: int
        """
        minValue = [float("inf")]*k
        minValue[0] = 0
        ans = -float("inf")
        total = 0
        for i, n in enumerate(nums):
            total += n
            length = (i+1)%k
            ans = max(ans, total-minValue[length])
            minValue[length] = min(minValue[length], total)
        return ans 
```

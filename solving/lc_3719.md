# Longest Balanced Subarray I

```java
class Solution {
    public int longestBalanced(int[] nums) {
        int ans = 0;
        for(int i=0; i<nums.length; i++){
            HashMap<Integer, Integer> evenFreq = new HashMap<>();
            HashMap<Integer, Integer> oddFreq = new HashMap<>();
            for(int j=i; j<nums.length; j++){
                if(nums[j]%2==0){
                    evenFreq.put(nums[j], evenFreq.getOrDefault(nums[j], 0) + 1);
                }else{
                    oddFreq.put(nums[j], oddFreq.getOrDefault(nums[j], 0) + 1);
                }
                if(evenFreq.size() == oddFreq.size()) ans = Math.max(ans, j - i + 1);
            }
            if(nums[i]%2==0){
                evenFreq.put(nums[i], evenFreq.get(nums[i])-1);
                if(evenFreq.get(nums[i]) == 0) evenFreq.remove(nums[i]);
            }else{
                oddFreq.put(nums[i], oddFreq.get(nums[i])-1);
                if(oddFreq.get(nums[i]) == 0) oddFreq.remove(nums[i]);
            }
        }
        return ans;
    }
}
```

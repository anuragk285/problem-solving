# Minimum Cost to Convert String I

```java
class Solution {
    public long minimumCost(String source, String target, char[] original, char[] changed, int[] cost) {
        List<List<int[]>> graph = new ArrayList<>();
        for(int i=0; i<26; i++) graph.add(new ArrayList<>());
        for(int i=0; i<original.length; i++){
            int u = original[i] - 'a';
            int v = changed[i] - 'a';
            int c = cost[i];
            graph.get(u).add(new int[]{v, c});
        }
        long minCost[][] = floyd(graph);
        long ans = 0;
        for(int i=0; i<source.length(); i++){
            int u = source.charAt(i)-'a';
            int v = target.charAt(i)-'a';
            if(u == v) continue;
            if(minCost[u][v] == Long.MAX_VALUE) return -1;
            ans += minCost[u][v];
        }
        return ans;
    }
    private long[][] floyd(List<List<int[]>> graph){
        long[][] ans = new long[26][26];
        for(int i=0; i<26; i++) Arrays.fill(ans[i], Long.MAX_VALUE);
        for(int i=0; i<26; i++){
            for(int[] neigh : graph.get(i)){
                int u = i, v = neigh[0], c = neigh[1];
                ans[u][v] = Math.min(ans[u][v], c);
            }
        }
        for(int k=0; k<26; k++){
            for(int u=0; u<26; u++){
                for(int v=0; v<26; v++){
                    long max = Long.MAX_VALUE;
                    if(ans[u][k] == max || ans[k][v] == max) continue;
                    ans[u][v] = Math.min(ans[u][v], ans[u][k] + ans[k][v]);
                }
            }
        }
        return ans;
    }
}
```

# Diagonal traverse

```python
class Solution(object):
    def findDiagonalOrder(self, mat):
        """
        :type mat: List[List[int]]
        :rtype: List[int]
        """
        m, n = len(mat), len(mat[0])
        ans = []
        i, j = 0, 0

        def move_down(i, j):
            while j >= 0 and i < m:
                ans.append(mat[i][j])
                i, j = i+1, j-1
            return i-1, j+1

        def move_up(i, j):
            while i >= 0 and j < n:
                ans.append(mat[i][j])
                i, j = i-1, j+1
            return i+1, j-1
        
        while True:

            if i == m-1 and j == n-1: break

            i, j = move_up(i, j)
            if i == 0 and j < n-1:
                j = j+1
            elif i < m and j == n-1:
                i += 1

            if i == m-1 and j == n-1: break

            i, j = move_down(i, j)
            if j == 0 and i < m-1:
                i += 1
            elif i == m-1 and j < n:
                j += 1
        ans.append(mat[m-1][n-1])
        return ans
```

# Degree of an Array

```java
class Solution {
    int[] nums;
    public int findShortestSubArray(int[] nums) {
        this.nums = nums;
        HashMap<Integer, Integer> freq = new HashMap<>();
        for(int n : nums) freq.put(n, freq.getOrDefault(n, 0)+1);
        int element, degree = -1;
        ArrayList<Integer> elements = new ArrayList<>();
        for(int k : freq.keySet()){
            if(degree == -1){
                elements.add(k);
                degree = freq.get(k);
            }else if(freq.get(k) > degree){
                elements.clear();
                element = k;
                elements.add(k);
                degree = freq.get(k);
            }
            else if(freq.get(k) == degree){
                elements.add(k);
            }
        }
        int ans = Integer.MAX_VALUE;
        for(int n : elements){
            int f = length(n, degree);
            //System.out.println(n+", subarray length: "+f);
            ans = Math.min(ans, f);
        }
        return ans;
    }
    private int length(int n, int degree){
        int start=0, end=0, i =0, count = 0;
        while(i<nums.length){
            if(nums[i] == n){
                if(count == 0) start = i;
                count++;
                if(count == degree) end = i;
            }
            i++;
        }
        return end - start + 1;
    }
}
```

# Ones and Zeroes

```java
class Solution {
    HashMap<List<Integer>, Integer> dp = new HashMap<>();
    String []strs;
    public int findMaxForm(String[] strs, int m, int n) {
        this.strs = strs;
        return dfs(0, m, n);
    } 
    public int dfs(int i, int m, int n){
        if(i >= this.strs.length) return 0;
        List<Integer> key = List.of(i, m, n);
        if(this.dp.containsKey(key)) return dp.get(key);
        int mCnt = count(strs[i], '0');
        int nCnt = count(strs[i], '1');
        this.dp.put(key, dfs(i+1, m, n));
        if(mCnt <= m && nCnt <= n){
            int max = Math.max(this.dp.get(key), 1+dfs(i+1, m-mCnt, n-nCnt));
            this.dp.put(key, max);
        }
        return this.dp.get(key);
    }
    public int count(String s, char key){
        int ans = 0;
        for(char c : s.toCharArray()) if(c == key) ans++;
        return ans;
    }
}
```

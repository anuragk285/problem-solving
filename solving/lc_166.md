# Fraction to recurring decimal

```python
class Solution(object):
    def fractionToDecimal(self, numerator, denominator):
        """
        :type numerator: int
        :type denominator: int
        :rtype: str
        """
        n, d = numerator, denominator
        quotient = ""
        recurring_start = -1
        negative = True if n<0 or d<0 else False
        n, d = abs(n), abs(d)
        indices = {}
        divisors = set()
        if n < d:
            quotient += '0'
        while n != 0:
            if n<d:
                if '.' not in quotient:
                    quotient += '.'
                n *= 10
            remainder = n%d 
            q = str(n//d)
            if remainder in divisors:
                recurring_start = indices[remainder]
                break
            else:
                divisors.add(remainder)
                quotient += q
                indices[remainder] = len(quotient)-1
            n = remainder
        if recurring_start != -1:
            quotient = quotient[:recurring_start] + "(" + quotient[recurring_start:] + ")"
        if negative:
            quotient = '-' + quotient
        return quotient                
```

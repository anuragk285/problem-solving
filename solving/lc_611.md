# Valid Triangle number

```python
class Solution(object):
    def triangleNumber(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        nums.sort()
        n = len(nums)
        count = 0
        for i in range(n-1, 1, -1):
            l, r = 0, i-1
            while l<r:
                if nums[l]+nums[r]>nums[i]:
                    count += (r-l)
                    r-=1
                else:
                    l+=1
        return count
```

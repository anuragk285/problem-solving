# Evaluate reverse polish notation

```python
class Solution(object):
    def evalRPN(self, tokens):
        """
        :type tokens: List[str]
        :rtype: int
        """
        def is_operand(c):
            if c == '+' or c == '-' or c == '*' or c == '/':
                return False
            return True
        
        st = [] # stores numbers

        for c in tokens:

            if is_operand(c):
                st.append(int(c))
            else:
                res = 0
                b = st.pop()
                a = st.pop()
                if c == '+':
                    res = a+b
                elif c == '-':
                    res = a-b
                elif c == '*':
                    res = a*b
                elif c == '/':
                    if a == 0:
                        res = 0
                    elif (a < 0 and b < 0) or (a > 0 and b > 0):
                        res = a//b
                    elif a < 0 and b > 0:
                        res = -(-a//b)
                    elif a > 0 and b < 0:
                        res = -(a//-b)
                st.append(res)
        return st.pop()
```

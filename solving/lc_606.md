# Construct String from binary tree 

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    String ans = "";
    public String tree2str(TreeNode root) {
        dfs(root);
        return this.ans;
    }
    public void dfs(TreeNode p){
        if(p == null) return;
        this.ans += Integer.toString(p.val);
        if(p.left == null && p.right == null) return;
        ans += "(";
        dfs(p.left);
        ans += ")";
        if(p.right != null){
            ans += "(";
            dfs(p.right);
            ans += ")";
        }
    }
}
```

# smallest palindromic rearrangement

```python
class Solution(object):
    def smallestPalindrome(self, s):
        """
        :type s: str
        :rtype: str
        """
        alpha = [0]*26
        for c in s : 
            alpha[ord(c)-ord('a')] += 1
        
        odd = -1
        ans = ""
        reverse_ans = ""
        for i in range(26):
            
            while alpha[i] >= 1:
                if alpha[i] == 1 :
                    odd = i
                    break
                if alpha[i] >= 2:
                    ans += chr(ord('a')+i)
                    reverse_ans += chr(ord('a')+i)
                    alpha[i] -= 2
        if odd != -1 : ans += chr(ord('a')+odd)
        ans = ans + reverse_ans[::-1]
        return ans     
```

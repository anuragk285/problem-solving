# Longest Palindrome

```java
class Solution {
    public int longestPalindrome(String s) {
        HashMap<Character, Integer> freq = new HashMap<>();
        for(char c : s.toCharArray()) freq.put(c, freq.getOrDefault(c, 0)+1);
        int ans =0;
        boolean odd = false;
        for(Map.Entry<Character, Integer> m : freq.entrySet()){
            int frq = m.getValue();
            if(frq%2 == 0) ans += frq;
            else{
                ans += frq - 1;
                odd = true;
            }
        }
        return ans + (odd ? 1 : 0);
    }
}
```

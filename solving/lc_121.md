# Best time to buy and sell stock 

```python
class Solution(object):
    def maxProfit(self, prices):
        """
        :type prices: List[int]
        :rtype: int
        """
        buy = [0]*len(prices)
        sell = [0]*len(prices)

        left_min, right_max = -1, -1

        for i in range(0, len(prices)):
            if left_min == -1: left_min = prices[i]
            else : left_min = min(left_min, prices[i])
            buy[i] =  left_min
        
        for i in range(len(prices)-1, -1, -1):
            if right_max == -1: right_max = prices[i]
            else : right_max = max(right_max, prices[i])
            sell[i] = right_max
        
        profit = 0
        for i in range(len(prices)):
            profit = max(profit, sell[i]-buy[i])
        
        return profit
```

# Dinf X-Sum of all K-Long Subarrays I

```python
class Solution(object):
    def findXSum(self, nums, k, x):
        """
        :type nums: List[int]
        :type k: int
        :type x: int
        :rtype: List[int]
        """
        ans = []
        for i in range(len(nums)-k+1):
            sum = 0
            freq = {}
            for j in range(i, i+k):
                sum += nums[j]
                freq[nums[j]] = freq.get(nums[j], 0) + 1
            if len(freq) < x:
                ans.append(sum)
            else:
                heap = []
                for key in freq:
                    heapq.heappush(heap, (-freq[key], -key))
                xsum = 0
                for _ in range(x):
                    f, n = heapq.heappop(heap)
                    n, f =  n*-1, f*-1 
                    xsum += n*f
                ans.append(xsum)
        return ans
```

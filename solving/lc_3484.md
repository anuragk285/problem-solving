# Design Spreadsheet

```python
class Spreadsheet(object):

    def __init__(self, rows):
        """
        :type rows: int
        """
        self.sheet = [[0]*26 for _ in range(rows)]

    def setCell(self, cell, value):
        """
        :type cell: str
        :type value: int
        :rtype: None
        """
        col = ord(cell[0]) - ord('A')
        row = int(cell[1:])
        self.sheet[row-1][col] = value
        

    def resetCell(self, cell):
        """
        :type cell: str
        :rtype: None
        """
        self.setCell(cell, 0)


    def getvalue(self, formula):
        if ord('A')<=ord(formula[0])<=ord('Z'):
            col = ord(formula[0])-ord('A')
            row = int(formula[1:])
            return self.sheet[row-1][col]
        else:
            return int(formula)
    def getValue(self, formula):
        """
        :type formula: str
        :rtype: int
        """
        a, b = formula.split('+')
        a = a[1:]
        return self.getvalue(a) + self.getvalue(b)
```

# minimum number of people to teach

```python
class Solution(object):
    def minimumTeachings(self, n, languages, friendships):
        """
        :type n: int
        :type languages: List[List[int]]
        :type friendships: List[List[int]]
        :rtype: int
        """
        sad_users = set()
        for u, v in friendships:
            common_lang = list(set(languages[u-1]) & set(languages[v-1]))
            if len(common_lang) > 0:
                continue
            else:
                sad_users.add(u)
                sad_users.add(v)
        
        all_lang = [0]*(n+1)
        for su in sad_users:
            for l in languages[su-1]:
                all_lang[l] += 1
        max_talk_lang = max(all_lang)

        return len(sad_users)-max_talk_lang
```

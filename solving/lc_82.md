# Remove Duplicates from Sorted List 2

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode deleteDuplicates(ListNode head) {
        if(head == null || head.next == null) return head;
        ListNode p = head, ansHead = null, lastNode = null;
        while(p != null && p.next != null){
            while(checkDuplicates(p)){
                while(checkDuplicates(p)) p = p.next;
                if(p != null) p = p.next;
            }
            if(ansHead == null){
                ansHead = p;
                lastNode = ansHead;
            }else{
                lastNode.next = p;
                lastNode = p;
            }
            if(p != null )p = p.next;
        }
        if(p != null) lastNode.next = p;
        return ansHead;
    }
    private boolean checkDuplicates(ListNode node){
        if(node == null || node.next == null) return false;
        if(node.val == node.next.val) return true;
        return false;
    }
}
```

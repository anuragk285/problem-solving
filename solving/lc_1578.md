# minimum time to make rope colourful

```python
class Solution(object):
    def minCost(self, colors, neededTime):
        """
        :type colors: str
        :type neededTime: List[int]
        :rtype: int
        """
        i = 0
        min_cost = 0
        while i<len(colors):
            c = colors[i]
            max_ = 0
            sum = 0
            while i<len(colors) and colors[i] == c:
                max_ = max(neededTime[i], max_)
                sum = sum+neededTime[i]
                i += 1
            if sum > max_:
                min_cost += sum - max_
        return min_cost
```

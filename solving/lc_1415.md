# The k-th Lexicographical String of All Happy Strings of Length n

```java
class Solution {
    int n=-1, k=-1, count=0;
    String ans = "";
    public String getHappyString(int n, int k) {
        this.n=n;
        this.k=k;
        if(generate(new LinkedList<Character>())) return ans;
        return ans;
    }
    private boolean generate(LinkedList<Character> s){
        if(s.size() == n){
            count++;
            if(count == k){
                for(char c : s) ans += c;
                return true;
            }
        }else{
            for(char c : "abc".toCharArray()){
                if(s.size() > 0 && s.getLast() == c) continue;
                s.addLast(c);
                if(generate(s)) return true;
                s.removeLast();
            }
        }
        return false;
    }
}
```

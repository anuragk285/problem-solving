# koko eating bananas

```python
class Solution(object):
    def minEatingSpeed(self, piles, h):
        """
        :type piles: List[int]
        :type h: int
        :rtype: int
        """
        low, high = 1, max(piles)
        ans = None
        while low <= high:
            k = (low+high)//2
            time = 0
            for p in piles:
                if p%k == 0: time += p//k
                else : time += (p//k) + 1
            if time <= h:
                if not ans : ans = k
                else : ans = min(ans, k)
                high = k-1
            else : low = k+1
        return ans
```

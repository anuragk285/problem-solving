# Serialize and deserialize Binary Tree

```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Codec:

    def serialize(self, root):
        """Encodes a tree to a single string.
        :type root: TreeNode
        :rtype: str
        """
        st = []
        s = ""
        p = root
        while p or st:
            if p:
                s += str(p.val)+","
                st.append(p)
                p =p.left
            else:
                s += "N,"
                p = st[-1]; st.pop()
                p = p.right
        return s+"N,"        

    def deserialize(self, data):
        """Decodes your encoded data to tree.
        :type data: str
        :rtype: TreeNode
        """
        values = data[:-1].split(',')
        root = None
        self.i = 0
        def recursion():
            if values[self.i] == 'N':
                self.i += 1
                return None
            else:
                new_node = TreeNode(int(values[self.i]))
                self.i += 1
                new_node.left = recursion()
                new_node.right = recursion()
                return new_node

        root = recursion()
        return root
```

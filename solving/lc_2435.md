# Paths in Matrix whose sum is Divisible by k

```java
class Solution {
    int[][] grid;
    int k;
    int ROWS, COLS;
    int[][][] cache;
    public int numberOfPaths(int[][] grid, int k) {
        this.grid = grid;
        this.k = k;
        this.ROWS = grid.length;
        this.COLS = grid[0].length;
        this.cache = new int[ROWS][COLS][k];
        for(int i=0; i<ROWS; i++) for(int j=0; j<COLS; j++) for(int x=0; x<k; x++) cache[i][j][x] = -1;
        return bt(0, 0, 0)% ((int)Math.pow(10, 9) + 7);
    }
    public int bt(int r, int c, int remainder){
        if(r == this.ROWS-1 && c == this.COLS-1){
            return (remainder + this.grid[r][c])%k == 0 ? 1 : 0;
        }else if(r == this.ROWS || c == this.COLS) return 0;
        else if(this.cache[r][c][remainder] > -1) return this.cache[r][c][remainder];
        else{
            int MOD = (int)Math.pow(10, 9) + 7;
            this.cache[r][c][remainder] = (
                bt(r+1, c, (remainder+grid[r][c])%k)%MOD + bt(r, c+1, (remainder+grid[r][c])%k)%MOD
            );
        }
        return cache[r][c][remainder];
    }

}
```

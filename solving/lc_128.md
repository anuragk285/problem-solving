# longest consecutive sequence

```python
class Solution(object):
    def longestConsecutive(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if len(nums) == 0 : return 0
        
        hash_set = set(nums)
        ans = 0
        nums = set(nums)
        for i in nums:

            if i-1 not in hash_set:
                length = 0
                while i+length in hash_set:
                    length += 1
                ans = max(ans, length)
        
        return ans
```

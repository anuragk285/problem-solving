# Subsets II

```java
class Solution {
    List<List<Integer>> ans = new ArrayList<>();
    int[] nums;
    public List<List<Integer>> subsetsWithDup(int[] nums) {
        this.nums = nums;
        backtrack(0, new ArrayList<Integer>());
        return this.ans;
    }
    public void backtrack(int i, ArrayList<Integer> subset){
        if(i == this.nums.length){
            Collections.sort(subset);
            if(!this.ans.contains(subset)) this.ans.add(new ArrayList<>(subset));
        }else{
            backtrack(i+1, subset);
            subset.add(this.nums[i]);
            backtrack(i+1, subset);
            subset.remove(Integer.valueOf(this.nums[i]));
        }
    }
}
```

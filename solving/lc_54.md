# Spiral Matrix

```java
class Solution {
    List<Integer> ans = new ArrayList<>();
    int rows =0, cols =0, ROWS, COLS;
    boolean[][] visited;
    int[][] matrix;
    public List<Integer> spiralOrder(int[][] matrix) {
        this.matrix = matrix;
        rows = matrix.length;
        cols = matrix[0].length;
        ROWS = rows;
        COLS = cols;
        visited = new boolean[rows][cols];
        int i=0, j=0;
        while(isValid(i, j) && !visited[i][j]){
            roundTheMatrix(i, j);
            i++;
            j++;
        }
        return ans;
    }
    private void roundTheMatrix(int r, int c){
        //To Right
        for(int j=c; j<c+cols; j++) addIfValid(r, j);

        //To Bottom
        for(int i=r; i<r+rows; i++) addIfValid(i, c+cols-1);

        //To Left
        for(int j=c+cols-1; j>=c; j--) addIfValid(r+rows-1, j);

        //To Top
        for(int i=r+rows-1; i>=r; i--) addIfValid(i, c);
        rows -= 2;
        cols -= 2;
    }

    private boolean addIfValid(int i, int j){
        if(!isValid(i, j)) return false;
        if(visited[i][j]) return false;
        ans.add(matrix[i][j]);
        visited[i][j] = true;
        return true;
    }

    private boolean isValid(int i, int j){
        if(i < 0 || i >= ROWS || j < 0 || j >= COLS) return false;
        return true;
    }
}
```

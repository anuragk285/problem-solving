# Reshape the matrix

```
class Solution {
    public int[][] matrixReshape(int[][] mat, int R, int C) {
        int[][] ans = new int[R][C];
        int r = 0, c = 0;
        if(mat.length * mat[0].length != R * C) return mat;
        for(int i=0; i<mat.length; i++){
            for(int j=0; j<mat[0].length; j++){
                if(c == C){
                    r = r+1;
                    c = 0;
                }
                ans[r][c] = mat[i][j];
                c++;
            }
        }
        return ans;
    }
}
```

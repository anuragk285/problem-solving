# Top k frequent elements

```c++
class Solution:
    def topKFrequent(self, nums, k):
        
        dic = {}

        for i in nums : 

            if i in dic :
                dic[i] += 1
            else : dic[i] = 1

        ans = []

        while k > 0 :

            max = 0; first = True
            for d in dic :
                if first : 
                    max = d
                    first = False
                elif dic[max] < dic[d] : max = d
            
            ans.append(max)
            dic.pop(max)
            k -= 1
        
        return ans
```

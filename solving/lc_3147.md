# Taking maximum energy from mystic dungeon

```python
class Solution(object):
    def maximumEnergy(self, energy, k):
        """
        :type energy: List[int]
        :type k: int
        :rtype: int
        """
        n = len(energy)
        self.dp = [False]*n
        self.ans = -float("inf")
        def bt(i):
            if i>=n:
                return 0
            elif self.dp[i] != False:
                return self.dp[i]
            else:
                self.dp[i] = bt(i+k) + energy[i]
            return self.dp[i]
        for i in range(n):
            bt(i)

        return max(self.dp)
```

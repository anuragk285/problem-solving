# Construct the Minimum Bitwise Array II

```java
class Solution {
    public int[] minBitwiseArray(List<Integer> nums) {
        int[] ans = new int[nums.size()];
        for(int i=0; i<nums.size(); i++){
            int y = nums.get(i);
            int k = trailingOnes(Integer.toBinaryString(y));
            int low = y - ((int)Math.pow(2, k)-1);
            int high = y - (int)Math.pow(2, k-1);
            for(int x=low; x<=high; x++){
                if((x | (x + 1)) == y) ans[i] = x;
                else ans[i] = -1;
            }
        }
        return ans;
    }
    private int trailingOnes(String s){
        int count = 0, i = s.length()-1;
        while(i>=0 && s.charAt(i) == '1'){
            count++;
            i--;
        }
        return count;
    }
}
```

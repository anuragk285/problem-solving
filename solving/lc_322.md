# Coin Change

```python
class Solution(object):
    def coinChange(self, coins, amount):
        """
        :type coins: List[int]
        :type amount: int
        :rtype: int
        """
        dp = [float("inf")]*(amount+1)
        dp[0] = 0
        for amt in range(1, amount+1):
            for c in coins:
                if amt - c >= 0:
                    dp[amt] = min(dp[amt], 1+dp[amt-c])
        return dp[amount] if dp[amount] != float("inf") else -1
```

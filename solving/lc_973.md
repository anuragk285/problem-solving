# K closest points to origin

```python
class Solution(object):
    def kClosest(self, points, k):
        """
        :type points: List[List[int]]
        :type k: int
        :rtype: List[List[int]]
        """
        heap = []
        for x, y in points:
            distance = (x**2 + y**2)**(0.5)
            heapq.heappush(heap, (distance, [x, y]))
        
        res = []
        for _ in range(k):
            distance, point = heapq.heappop(heap)
            res.append(point)
    
        return res
```

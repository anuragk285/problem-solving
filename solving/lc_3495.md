# minimum operations to make array elements to zero

```python
class Solution(object):
    def minOperations(self, queries):
        """
        :type queries: List[List[int]]
        :rtype: int
        """
        def get(num):
            if num == 0 : return 0
            i = 1
            base = 1
            res = 0
            while True:
                l = 4**(i-1)
                r = (4**i)-1
                if l > num: break
                res += (min(r, num)-l + 1)*base
                i += 1
                base += 1
            return res
                
        count = 0
        for l, r in queries:
            count += (get(r)-get(l-1) + 1)//2
        return count
```

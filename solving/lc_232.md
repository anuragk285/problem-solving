# Implement Queue using stacks

```python
class MyQueue(object):

    def __init__(self):
        self.adder = []
        self.remover = []

    def push(self, x):
        """
        :type x: int
        :rtype: None
        """
        self.adder.append(x)
        

    def pop(self):
        """
        :rtype: int
        """
        if not self.remover:
            while self.adder:
                self.remover.append(self.adder.pop())

        return self.remover.pop()

    def peek(self):
        """
        :rtype: int
        """
        if self.remover:
            return self.remover[-1]
        else:
            return self.adder[0]

    def empty(self):
        """
        :rtype: bool
        """
        return False if self.adder or self.remover else True
```

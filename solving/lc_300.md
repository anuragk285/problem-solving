# Longest Increasing Subsequence

```python
class Solution(object):
    def lengthOfLIS(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        dp = [0]*n
        for i in range(n-1, -1, -1):
            if i == n-1:
                dp[i] = 1
            else:
                max_ = 0
                for j in range(i+1, n):
                    if nums[j] > nums[i]:
                        max_ = max(max_, dp[j])
                dp[i] = max(1, max_+1)
        return max(dp)       
```

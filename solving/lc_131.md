# palindrome partitioning

```python
class Solution(object):
    def partition(self, s):
        """
        :type s: str
        :rtype: List[List[str]]
        """
        self.res = []
        def bt(i, part):
            if i >= len(s):
                self.res.append(part[:])
            else:
                for j in range(i, len(s)):
                    if s[i:j+1] == (s[i:j+1])[::-1]:
                        part.append(s[i:j+1])
                        bt(j+1, part)
                        part.pop()
        bt(0, [])
        return self.res
```

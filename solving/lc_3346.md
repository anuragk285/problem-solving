# Maximum frequency of an element after performing operations I

```python
class Solution(object):
    def maxFrequency(self, nums, k, numOperations):
        """
        :type nums: List[int]
        :type k: int
        :type numOperations: int
        :rtype: int
        """
        nums.sort()
        self.freq = {}
        ans = 0
        for i in nums:
            self.freq[i] = self.freq.get(i, 0) + 1
        def getIndex(n):
            left = 0
            right = len(nums)
            while left < right:
                mid = (left + right) // 2
                if nums[mid] < n:
                    left = mid + 1
                else:
                    right = mid
            return left
        def maxFreq(n):
            left = getIndex(n-k)
            right = getIndex(n+k+1)
            operations = right - left - self.freq.get(n, 0)
            return min(operations, numOperations) + self.freq.get(n, 0)
        max_ = nums[-1]
        for i in range(1, max_+1):
            ans = max(ans, maxFreq(i))
        return ans
```

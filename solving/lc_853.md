# Car Fleet 

```python
class Solution(object):
    def carFleet(self, target, position, speed):
        """
        :type target: int
        :type position: List[int]
        :type speed: List[int]
        :rtype: int
        """
        pairs = [[p, s] for p, s in zip(position, speed)]

        st = []

        for p, s in sorted(pairs)[::-1]:
            st.append((float(target-p)/s))
            if len(st) >= 2 and st[-1] <= st[-2]:
                st.pop()
            #print(st)
        
        return len(st)
```

# House Robber II

```python
class Solution(object):
    def rob(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if len(nums) == 1:
            return nums[0]
        ans = 0
        rob1, rob2 = 0, 0
        for n in nums[:-1]:
            temp = max(n+rob1, rob2)
            rob1 = rob2
            rob2 = temp
        ans = max(ans, rob2)
        rob1, rob2 = 0, 0
        for n in nums[1:]:
            temp = max(n+rob1, rob2)
            rob1 = rob2
            rob2 = temp
        return max(ans, rob2)
```

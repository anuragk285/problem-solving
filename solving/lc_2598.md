# Smallest Missing non-negative Integer after operations

```python
class Solution(object):
    def findSmallestInteger(self, nums, value):
        """
        :type nums: List[int]
        :type value: int
        :rtype: int
        """
        freq = {i : 0 for i in range(value)}
        for n in nums:
            remainder = ((n % value) + value) % value
            freq[remainder] = freq[remainder] + 1
        
        remainder = 0
        count = float("inf")
        for i in range(value):
            cnt = freq[i]
            if count > cnt:
                count = cnt
                remainder = i

        return value * freq[remainder] + remainder
```

from sortedcontainers import SortedSet
class TaskManager(object):

    def __init__(self, tasks):
        """
        :type tasks: List[List[int]]
        """
        self.tasks = {}
        self.prior_tasks = SortedSet()
        for userId, taskId, priority in tasks:
            self.tasks[taskId] = [userId, priority]
            self.prior_tasks.add((-priority, -taskId))


    def add(self, userId, taskId, priority):
        """
        :type userId: int
        :type taskId: int
        :type priority: int
        :rtype: None
        """
        self.tasks[taskId] = [userId, priority]
        self.prior_tasks.add((-priority, -taskId))
        

    def edit(self, taskId, newPriority):
        """
        :type taskId: int
        :type newPriority: int
        :rtype: None
        """
        old_priority = self.tasks[taskId][1]
        self.tasks[taskId][1] = newPriority
        self.prior_tasks.remove((-old_priority, -taskId))
        self.prior_tasks.add((-newPriority, -taskId))

    def rmv(self, taskId):
        """
        :type taskId: int
        :rtype: None
        """
        userId, priority = self.tasks[taskId]
        del self.tasks[taskId]
        self.prior_tasks.remove((-priority, -taskId))
        

    def execTop(self):
        """
        :rtype: int
        """
        if not self.prior_tasks: return -1
        priority, taskId = self.prior_tasks[0]
        userId, p = self.tasks[-taskId]
        del self.tasks[-taskId]
        self.prior_tasks.remove((priority, taskId))
        return userId

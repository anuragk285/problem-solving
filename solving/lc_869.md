# reordered power of 2

```python
class Solution(object):
    def reorderedPowerOf2(self, n):
        """
        :type n: int
        :rtype: bool
        """
        def count_digits(n):
            digits = [0]*10
            while n>0:
                digits[n%10] += 1
                n //= 10
            return digits

        original_digits = count_digits(n)
        
        for i in range(31):
            num = 1 << i
            if count_digits(num) == original_digits:
                return True
        
        return False
```

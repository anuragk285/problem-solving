# Circular Array Loop

```java
class Solution {
    public boolean circularArrayLoop(int[] nums) {
        int n = nums.length;
        for(int i=0; i<nums.length; i++){
            HashSet<Integer> visited = new HashSet<>();
            int j = i;
            while(!visited.contains(j)){
                visited.add(j);
                //System.out.print(j+", ");
                j = (((j + nums[j]) % n)  + n) % n;
            }   
            int length = 0, idx = j;
            boolean positive = false, negative = false, cycle = true;
            do{
                if(nums[idx] > 0) positive = true;
                else negative = true;
                if(positive && negative){
                    cycle = false;
                    break;
                }
                idx = (((idx + nums[idx])%n) + n) %n;
                length++;
            }while(idx != j);
            //System.out.println();
            if(length > 1 && cycle) return true;
        }
        return false;
    }
}
```

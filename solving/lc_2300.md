# Successful pairs of spells and potions

```python
class Solution(object):
    def successfulPairs(self, spells, potions, success):
        """
        :type spells: List[int]
        :type potions: List[int]
        :type success: int
        :rtype: List[int]
        """
        n = len(potions)
        potions.sort()
        res = []
        for s in spells:
            p = floor(success) / s
            i = bisect.bisect_left(potions, ceil(p))
            if i == n:
                res.append(0)
            else:
                res.append(n-i)
        return res
```

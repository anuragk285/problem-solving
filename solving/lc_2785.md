# Sort vowels in a string

```python
class Solution(object):
    def sortVowels(self, s):
        """
        :type s: str
        :rtype: str
        """
        vowels = ['a', 'e', 'i', 'o', 'u']
        heap = [] # (ord(char), char)

        for i, c in enumerate(s):   
            if c.lower() in vowels:
                heapq.heappush(heap, (ord(c), c))
        
        t = ""
        for i, c in enumerate(s):
            if c.lower() not in vowels:
                t += c
            else:
                _ord, ch = heapq.heappop(heap)
                t += ch
        
        return t
```

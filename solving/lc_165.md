# Compare version numbers

```python
class Solution(object):
    def compareVersion(self, version1, version2):
        """
        :type version1: str
        :type version2: str
        :rtype: int
        """
        rev1, rev2 = [], []
        i, j = 0, 0
        while i<len(version1):
            r1 = 0
            while i<len(version1) and version1[i] != '.':
                r1 = r1*10 + int(version1[i])
                i += 1
            rev1.append(r1)
            i += 1
        while j<len(version2):
            r2 = 0
            while j<len(version2) and version2[j] != '.':
                r2 = r2*10 + int(version2[j])
                j += 1
            rev2.append(r2)
            j += 1
        i, j = 0, 0
        while i<len(rev1) and j<len(rev2):
            if rev1[i] < rev2[j]:
                return -1
            elif rev1[i] > rev2[j]: 
                return 1
            i, j = i+1, j+1
        while j<len(rev2):
            if rev2[j]>0:
                return -1
            j += 1
        while i<len(rev1):
            if rev1[i]>0:
                return 1
            i += 1 
        return 0
```

# Trapping rain water

```python
class Solution(object):
    def trap(self, height):
        """
        :type height: List[int]
        :rtype: int
        """
        bounds = [[0,0] for _ in range(len(height))]
        max_height = 0
        left_bound = 0
        for i in range(1, len(height)):
            bounds[i][0] = max(max_height, height[i-1])
            max_height = max(max_height, bounds[i][0])

        max_height = 0
        for i in range(len(height)-2, -1, -1):
            bounds[i][1] = max(max_height, height[i+1])
            max_height = max(max_height, bounds[i][1])

        water = 0
        for i in range(1, len(height)-1):
            if not height[i] > min(bounds[i][0], bounds[i][1]):
                water += min(bounds[i][0], bounds[i][1])-height[i]
        
        return water
```

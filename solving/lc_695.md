# Max Area of Island

```java
class Solution {
    static class Pair{
        int i, j;
        Pair(int i, int j){
            this.i = i;
            this. j = j;
        }
    }
    int ROWS, COLS;
    int[][] grid;
    boolean[][] visited;
    public int maxAreaOfIsland(int[][] g) {
        this.grid = g;
        int maxArea = 0;
        ROWS = grid.length;
        COLS = grid[0].length;
        visited = new boolean[ROWS][COLS];
        for(int i=0; i<ROWS; i++){
            for(int j=0; j<COLS; j++){
                if(grid[i][j] == 1 && !visited[i][j]){
                    visited[i][j] = true;
                    int area = getArea(i, j);
                    //System.out.println(area);
                    maxArea = Math.max(maxArea, area);
                }
            }
        }
        return maxArea;
    }
    private int getArea(int r, int c){
        Queue<Pair> queue = new LinkedList<>();
        queue.add(new Pair(r, c));
        int area = 0;
        while(!queue.isEmpty()){
            area++;
            Pair p = queue.poll();
            int i = p.i, j = p.j;
            int[] nRows = {i-1, i, i+1, i};
            int[] nCols = {j, j+1, j, j-1};
            for(int idx=0; idx<4; idx++){
                r = nRows[idx];
                c = nCols[idx];
                if(isValid(r, c) && grid[r][c] == 1 && !visited[r][c]){
                    visited[r][c] = true;
                    queue.add(new Pair(r, c));
                }
            }
        }
        return area;
    }
    private boolean isValid(int i, int j){
        if(i<0 || i>=ROWS || j<0 || j>=COLS) return false;
        return true;
    }
}
```

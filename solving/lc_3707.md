# Equal score substrings

```python
class Solution(object):
    def scoreBalance(self, s):
        """
        :type s: str
        :rtype: bool
        """
        total_score = 0
        def score(c):
            return ord(c)-ord('a')+1
        
        for c in s:
            total_score += score(c)

        left_score = 0
        for i in range(len(s)):
            left_score += score(s[i])
            if left_score == total_score-left_score:
                return True

        return False
```

# Smallest absent positive greater than average

```python
class Solution(object):
    def smallestAbsent(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        avg = sum(nums)/len(nums)
        nums.sort()
        arr = [0]*(101)
        for i in nums:
            if i > 0:
                arr[i] = 1
        for i in range(1, 101):
            if i>avg:
                if arr[i] == 0: 
                    return i
        return max(nums)+1
```

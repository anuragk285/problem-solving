# Task Schedular

```python
class Solution(object):
    def leastInterval(self, tasks, n):
        """
        :type tasks: List[str]
        :type n: int
        :rtype: int
        """
        alpha = [0]*26
        for t in tasks:
            alpha[ord(t)-ord('A')] += 1
        heap = []
        for freq in alpha:
            if freq>0:
                heapq.heappush(heap, -freq)
        q = deque()
        time = 0
        while heap or q:
            time += 1
            if heap:
                val = 1 + heapq.heappop(heap)
                if val != 0:
                    q.append((val, time+n))
            if q and q[0][1] == time:
                v = q.popleft()
                heapq.heappush(heap, v[0])
        return time
```

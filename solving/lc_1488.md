# Avoid Flood in the city

```python
class Solution(object):
    def avoidFlood(self, rains):
        """
        :type rains: List[int]
        :rtype: List[int]
        """
        dry_days = []
        filled_lakes = dict()
        res = [-1]*len(rains)
        for i, r in enumerate(rains):
            if r == 0:
                dry_days.append(i)
                res[i] = 1
            else:
                if r in filled_lakes:
                    index = -1
                    for d in dry_days:
                        if d > filled_lakes[r]:
                            index = d
                            break
                    if index == -1:
                        return []
                    res[index] = r
                    dry_days.remove(index)
                    filled_lakes[r] = i
                else:
                    filled_lakes[r] = i
        return res
```

# Palindromic substrings

```python
class Solution(object):
    def countSubstrings(self, s):
        """
        :type s: str
        :rtype: int
        """
        def count_palindromes(i, j):
            count = 0
            while i>=0 and j<len(s):
                if s[i] == s[j]:
                    i, j = i-1, j+1
                    count += 1
                else:
                    break
            return count
        ans = 0
        for i in range(len(s)):
            odd_len = count_palindromes(i-1, i+1)
            even_len = count_palindromes(i, i+1)
            ans += 1 + odd_len + even_len
        
        return ans
```

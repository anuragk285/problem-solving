# Print words vertically

```python
class Solution(object):
    def printVertically(self, s):
        """
        :type s: str
        :rtype: List[str]
        """
        words = list(s.split(" "))
        min_size = -float("inf")
        for word in words: 
            min_size = max(min_size, len(word))
        res = []
        def remove_trailing_spaces(s):
            i = len(s)-1
            while s[i] == " ":
                i -= 1
            return s[:i+1]
        for i in range(min_size):
            res_word = ""
            for word in words:
                if i>=len(word):
                    res_word += " "
                else:
                    res_word += word[i]
            res.append(remove_trailing_spaces(res_word))
        return res
```

# lexicographical nunmbers

```c++
class Solution {
public:
    void dfs(vector<int>& nums, int n, int curr){

        if(curr > n) return;

        nums.push_back(curr);

        for(int i=0; i<=9; i++){

            if(curr*10 + i > n) return;
            dfs(nums, n, curr*10 + i);
        }
    }

    vector<int> lexicalOrder(int n) {
        
        vector<int> nums;

        for(int i=0; i<=9; i++) dfs(nums, n, i);

        return nums;
    }
};
```

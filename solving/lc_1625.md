# Lexicographically smallest strings after applying operations

```python
from collections import deque
class Solution(object):
    def findLexSmallestString(self, s, a, b):
        """
        :type s: str
        :type a: int
        :type b: int
        :rtype: str
        """
        visited = set()
        que = deque()
        ans = s
        visited.add(s)
        que.append(s)
        while que:
            n = que.popleft()
            ans = min(ans, n)
            n1 = ""
            for i, c in enumerate(n):
                new_digit = (ord(c)-ord('0') + a) % 10
                n1 += c if i%2==0 else str(new_digit)
            if n1 not in visited:
                visited.add(n1)
                que.append(n1)
            n2 = n[-b:] + n[:-b] 
            if n2 not in visited:
                visited.add(n2)
                que.append(n2)
        return ans
```

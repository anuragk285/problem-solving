# Binary tree maximum path sum

```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def maxPathSum(self, root):
        """
        :type root: Optional[TreeNode]
        :rtype: int
        """
        self.max_sum = root.val

        def get_sum(p):

            if not p : return 0

            left_sum = get_sum(p.left)
            right_sum = get_sum(p.right)

            left_sum = max(left_sum, 0)
            right_sum = max(right_sum, 0)
            mid_sum = left_sum + p.val + right_sum
            self.max_sum = max(self.max_sum, left_sum + p.val + right_sum)

            return p.val + max(left_sum, right_sum)
        
        get_sum(root)
        return self.max_sum
```

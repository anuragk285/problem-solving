# Sort integers by the power value

```python
class Solution(object):
    def getKth(self, lo, hi, k):
        """
        :type lo: int
        :type hi: int
        :type k: int
        :rtype: int
        """
        
        def power(n):

            steps = 0

            while n != 1 : 

                if n%2 == 0 : n = n//2

                else : n = (n*3)+1

                steps += 1
            return steps

        dic = {}

        for i in range(lo, hi+1):
            dic[i] = power(i)
        
        sorted_list = sorted(dic.items(), key = lambda item : (item[1], item[0]))

        return sorted_list[k-1][0]

```

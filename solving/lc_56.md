# Merge Intervals

```java
class Solution {
    public int[][] merge(int[][] intervals) {
        ArrayList<int[]> list = new ArrayList<>();
        for(int[] i : intervals) list.add(i);
        list.sort((a, b) -> Integer.compare(a[0], b[0]));
        ArrayList<int[]> ans = new ArrayList<>();
        for(int i=0; i<list.size(); i++){
            int[] interval = list.get(i);
            if(ans.size() == 0) ans.add(interval);
            else if(ans.get(ans.size()-1)[1] >= interval[0]){
                ans.get(ans.size()-1)[1] = Math.max(ans.get(ans.size()-1)[1], interval[1]);
            }
            else ans.add(interval);
        }
        int[][] mergedList = new int[ans.size()][2];
        for(int i=0; i<ans.size(); i++){
            mergedList[i][0] = ans.get(i)[0];
            mergedList[i][1] = ans.get(i)[1];
        }
        return mergedList;
    }
}
```

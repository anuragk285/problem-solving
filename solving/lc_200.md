# Number of Islands

```java
class Solution {
    class Pair{
        int i, j;
        Pair(int i, int j){
            this.i = i;
            this.j = j;
        }
    }
    char[][] grid;
    boolean[][] visited;
    int ROWS, COLS;
    public int numIslands(char[][] g) {
        this.grid = g;
        ROWS = g.length;
        COLS = g[0].length;
        visited = new boolean[ROWS][COLS];
        int ans = 0;
        for(int i=0; i<ROWS; i++){
            for(int j=0; j<COLS; j++){
                if(grid[i][j] == '1' && !visited[i][j]){
                    visited[i][j] = true;
                    ans++;
                    BFS(i, j);
                }
            }
        }
        return ans;
    }
    private void BFS(int R, int C){
        Queue<Pair> queue = new LinkedList<>();
        queue.add(new Pair(R, C));
        while(!queue.isEmpty()){
            Pair p = queue.poll();
            int i = p.i, j = p.j;
            int[] nRows = {i-1, i, i+1, i};
            int[] nCols = {j, j+1, j, j-1};
            for(int idx=0; idx<4; idx++){
                int r = nRows[idx];
                int c = nCols[idx];
                if(isValid(r, c) && grid[i][j] == '1' && !visited[r][c]){
                    queue.add(new Pair(r, c));
                    visited[r][c] = true;
                }
            }
        }
    }
    private boolean isValid(int i, int j){
        if(i < 0 || i>=ROWS || j<0 || j>=COLS) return false;
        return true;
    }
}
```

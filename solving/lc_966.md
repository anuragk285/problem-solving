# Vowel spell checker

```python
class Solution(object):
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        vowels = set("aeiou")
        exact = set(wordlist)
        case_map = {}
        vowel_map = {}
        res = []

        def devowel(s):
            res = ""
            for c in s.lower():
                res += (c if c not in vowels else ".")
            return res

        for w in wordlist:
            lower = w.lower()
            dvow = devowel(w)

            if lower not in case_map:
                case_map[lower] = w
            if dvow not in vowel_map:
                vowel_map[dvow] = w
        
        for q in queries:
            if q in exact:
                res.append(q)
            elif q.lower() in case_map:
                res.append(case_map[q.lower()])
            elif devowel(q) in vowel_map:
                res.append(vowel_map[devowel(q)])
            else:
                res.append("")  

        return res
```

# Longest Mountain in Array

```java
class Solution {
    int[] arr;
    public int longestMountain(int[] arr) {
        this.arr = arr;
        int ans = 0;
        for(int i=0; i<arr.length; i++){
            int dist = getDistance(i);
            //System.out.println("index: "+i+", Dist: "+dist);
            ans = Math.max(ans, dist);
        }
        return ans;
    }
    private int getDistance(int idx){
        int dist = 0, i = idx;
        if(idx+1>=arr.length || idx-1<0) return 0;
        int left_dist =0, right_dist = 0;
        while(idx-left_dist-1>=0 && arr[idx-left_dist-1] < arr[idx-left_dist]) left_dist++;
        while(idx+right_dist+1<arr.length && arr[idx+right_dist] > arr[idx+right_dist+1]) right_dist++;
        if(left_dist == 0 || right_dist == 0) return 0;
        return left_dist + 1 + right_dist;
    }
}
```

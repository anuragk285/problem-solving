# Maximum product of splitted binary tree

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    long tSum = 0, ans = 0;
    long mod = 1_000_000_007;
    public int maxProduct(TreeNode root) {
        tSum = totalSum(root);
        TreeNode n = root;
        long leftSum = subTreeSum(n.left);
        long leftSplit = ((tSum - leftSum) * leftSum);
        long rightSum =  subTreeSum(n.right);
        long rightSplit = ((tSum - rightSum) * rightSum);
        ans = Math.max(ans, leftSplit);
        ans = Math.max(ans, rightSplit);
        return (int)(ans%mod);
    }
    private long totalSum(TreeNode n){
        if(n == null) return 0;
        return totalSum(n.left) + totalSum(n.right) + n.val;
    }
    private long subTreeSum(TreeNode n){
        if(n == null) return 0;
        long leftSum = subTreeSum(n.left);
        long leftSplit = ((tSum - leftSum) * leftSum);
        long rightSum =  subTreeSum(n.right);
        long rightSplit = ((tSum - rightSum) * rightSum);
        ans = Math.max(ans, leftSplit);
        ans = Math.max(ans, rightSplit);
        return leftSum + n.val + rightSum;
    }
}
```

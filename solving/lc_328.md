# Odd even linked list

```python
# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution(object):
    def oddEvenList(self, head):
        """
        :type head: Optional[ListNode]
        :rtype: Optional[ListNode]
        """

        if not head or not head.next : return head

        index = 1
        even_head = None; even_last_node = None
        odd_head = None; odd_last_node = None

        p = head

        while p != None : 

            if index%2 == 0:

                if even_head == None :
                    even_head = ListNode(p.val)
                    even_last_node = even_head
                
                else :
                    even_last_node.next = p
                    even_last_node = p
            
            else :

                if odd_head == None :
                    odd_head = ListNode(p.val)
                    odd_last_node = odd_head
                
                else :
                    odd_last_node.next = p
                    odd_last_node = p
            index += 1
            p = p.next
        
        odd_last_node.next = even_head
        even_last_node.next = None

        return odd_head
```

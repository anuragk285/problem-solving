# Count Special Triplets

```java
class Solution {
    public int specialTriplets(int[] nums) {
        HashMap<Integer, Integer> right = new HashMap<>();
        for(int n : nums) right.put(n, right.getOrDefault(n, 0)+1);
        HashMap<Integer, Integer> left = new HashMap();
        long ans = 0, mod = 1_000_000_007L;
        for(int j=0; j<nums.length; j++){
            int x = nums[j];
            long twice = 2L * x;
            right.put(x, right.get(x)-1);
            int rightCount = right.getOrDefault((int)twice, 0);
            int leftCount = left.getOrDefault((int)twice, 0);
            ans = (ans + ((long)leftCount * rightCount)%mod)%mod;
            left.put(nums[j], left.getOrDefault(nums[j], 0)+1);
        }
        return (int)ans;
    }
}
```

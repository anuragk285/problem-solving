# search in a 2D matrix

```python
class Solution(object):
    def searchMatrix(self, matrix, target):
        """
        :type matrix: List[List[int]]
        :type target: int
        :rtype: bool
        """
        
        row_index = -1
        low, high = 0, len(matrix)-1
        mid = (low+high)//2

        while low<=high:

            mid = (low+high)//2
            print(mid)
            if mid == len(matrix)-1:
                if matrix[mid][0] <= target:
                    row_index = mid
                    break
            else:
                if matrix[mid][0] <= target and matrix[mid+1][0] > target:
                    row_index = mid
                    break
            
            if matrix[mid][0] < target:
                low = mid+1
            
            elif matrix[mid][0] > target:
                high = mid-1
        
        print(row_index)
        if row_index != -1:

            low, high = 0, len(matrix[row_index])-1
            mid = (low+high)//2

            while low<=high:
                mid = (low+high)//2
                print(mid)
                if matrix[row_index][mid] == target:
                    return True
                elif matrix[row_index][mid] < target:
                    low = mid+1
                else: high = mid-1
        
        return False
```

# find the number of ways to place people I

```python 
class Solution(object):
    def numberOfPairs(self, points):
        """
        :type points: List[List[int]]
        :rtype: int
        """
        count = 0
        for x, y in points:
            for i, j in points:
                if i == x and j == y: continue
                if i < x or j > y : continue
                disturbance = False
                for k, l in points:
                    if (k == i and l == j) or (k == x and l == y): continue
                    if x <= k <= i and y >= l >= j:
                        disturbance = True
                        break
                if not disturbance: count += 1
        return count
```
